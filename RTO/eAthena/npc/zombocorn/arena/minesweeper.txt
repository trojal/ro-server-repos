		mes "Please input the amount of mines to be planted on the field.";

		mes "[Time remaining: " + (5-$@MSMNTM) + ":00]";

		mes "[Room Controller]";
		mes "Firstly, the game is a 9x9 grid of tiles (porings).  That's 81 tiles.  There are a certain number of hidden mines on the field.";

		mes "[Room Controller]";
		mes "The goal of the game is to 'Flag' all the mines.  If you click on a tile and there is a mine there, game over!";
		next;
		mes "[Room Controller]";
		mes "When you click on a tile, the tile will be flagged, and turn blue.  If you click on a flagged tile, it will open up.  If there's nothing there, a chain reaction will start.  If there's a number there, it'll show the number.  If there's a mine there, game over!";
		next;
		mes "[Room Controller]";
		mes "Be careful!  You only have ONE chance to flag a tile!  No flagging and unflagging here.";
		next;
		mes "[Room Controller]";
		if ($MSWindow == 0) mes "Now about the tiles.  Yellow means the tile has not been clicked on yet.  Blue means the tile is flagged.  And if red, it will have a waiting room window with a number inside.  Say, if there is a red tile with a '5' in the waiting room, then that means there are 5 mines within a 3x3 box centered on that tile.";
		if ($MSWindow == 1) mes "Now about the tiles.  Yellow means the tile has not been clicked on yet.  Blue means the tile is flagged.  And if red, the npc's name will be the number of that tile.  Say, if there is a red tile with a '5' in the waiting room, then that means there are 5 mines within a 3x3 box centered on that tile.";
		if ($MSWindow == 2) mes "Now about the tiles.  Yellow means the tile has not been clicked on yet.  Blue means the tile is flagged.  And if red, it will have a waiting room window with a number inside and a waiting room with the number in there too.  Say, if there is a red tile with a '5' in the waiting room, then that means there are 5 mines within a 3x3 box centered on that tile.";
		next;
		mes "[Room Controller]";
		mes "You get points for clicking on a tile that has a number on it.  You have to finish the game to get any kind of reward though!";
		next;
		mes "[Room Controller]";
		mes "Reward is based on the amount of points you accumulate from opening numbered tiles, and how many mines there are on the field.";
		next;
		mes "[Room Controller]";
		mes "This is also a minigame you can play by yourself.  And that's about it!";
		next;
		goto M_Menu;

		mes "Alright.";
		close;
	M_Warpout:
		mes "[Room Controller]";
		mes "The game is in progress, are you sure you want to leave?";
		next;
		menu "No, I'll stay",M_Wait,"Yes.",M_Exit;

		if (@MSpt > 0) set @MSpt,0;
		close2;

		end;

		mes "[Room Controller]";
		mes "Hello "+strcharinfo(0)+", this room has ended and is in a cooldown phase.  The room will be open again in a short while.";
		mes " ";

		set $@MSSETM,((getnpctimer(0,"Minesweeper1") / 2000) % 60);
		if ($@MSSETM == 0 ) goto M_NewMin2;
		if ($@MSSETM >= 51 ) goto M_Addzero2;
		mes "[Time remaining: " + (2-$@MSMNTM) + ":" + (60-$@MSSETM) + "]"; // Counts down, this is the base timer
		goto M_NotReady2;
	M_NewMin2:
		mes "[Time remaining: " + (3-$@MSMNTM) + ":00]"; // Keeps the second from starting at 60
		goto M_NotReady2;
	M_Addzero2:
		mes "[Time remaining: " + (2-$@MSMNTM) + ":0" + (60-$@MSSETM) + "]"; // Adds a 0 infront of 1-9

		next;
		if ($@MSLose == 1) goto M_NotReady4;
		if (@MSpt > 0) goto M_NotReady3;

		if (getgmlevel() > 79) menu "Wait in room.",M_Wait,"Control Panel",M_Control,"Leave Room",M_Exit;
		if (getgmlevel() < 80) menu "Wait in room.",M_Wait,"Leave Room",M_Exit;
	M_NotReady3:
		if (getgmlevel() > 79) menu "Wait in room.",M_Wait,"Control Panel",M_Control,"Cash in",M_Cashin;
		if (getgmlevel() < 80) menu "Wait in room.",M_Wait,"Cash in",M_Cashin;

		mes "[Room Controller]";
		mes "Alright.... lets see how you did.";
		mes " ";
		mes " ";
		mes " ";
		mes "Total Points : "+@MSpt;
		if ($MSRewType == 0) mes "Kafra Points Earned : "+((@MSpt*$MSReward)+($@MSAmount*$@MSAmount)/$MSReward)+" KP";
		if ($MSRewType == 1) mes "Zeny Earned : "+((((@MSpt*$MSReward)+($@MSAmount*$@MSAmount)*50)/10)/$MSReward)+"z";
		next;
		mes "[Room Controller]";
		mes "Thank you for playing.";
		if ($MSRewType == 0) set RESRVPTS,RESRVPTS + ((@MSpt*$MSReward)+($@MSAmount*$@MSAmount)/$MSReward);
		if ($MSRewType == 1) set Zeny,Zeny + ((@MSpt*$MSReward)+($@MSAmount*$@MSAmount)*50/10/$MSReward);
		if ($MSRewType == 0) dispbottom "[You have "+RESRVPTS+" Kafra Reserve Points]";
		set @MSpt,0;
		next;
		mes "[Room Controller]";
		mes "Do you wish to stay here and wait for the room to become available, or return to town?";
		next;
		menu "Wait in room.",M_Wait,"Leave Room",M_Exit;
	L_Timer:
		set $@MSCD,1;
		setnpctimer 0; //safeguard
		StartNPCTimer "Minesweeper1";
		set $@MSOn,1;
		set $@MSTitle$,"- Game in Progress";
		mapannounce "minesweep",strcharinfo(0) + " has started the countdown!  There are "+$@MSAmount+" mines on the field!  You have a 5 minute time limit!",1;
		set $@MSCount,1;
		cleararray $@MSState[0],0,82;
		cleararray $@MSMines[0],0,82;
		cleararray $@MSField[0],0,82;
		goto L_Start;

		if ($@MSCount == 0) end;
		mapannounce "minesweep","[10]",1;
		end;
	OnTimer2000:
		if ($@MSCount == 0) end;
		mapannounce "minesweep","[9]",1;
		end;
	OnTimer3000:
		if ($@MSCount == 0) end;
		mapannounce "minesweep","[8]",1;
		end;

		if ($@MSCount == 0) end;
		mapannounce "minesweep","[7]",1;
		end;

		mapannounce "minesweep","[6]",1;
		end;

		if ($@MSCount == 0) end;
		mapannounce "minesweep","[5]",1;
		end;

		mapannounce "minesweep","[4]",1;

		mapannounce "minesweep","[2]",1;
		end;
	OnTimer10000:
		if ($@MSCount == 0) end;
		mapannounce "minesweep","[1]",1;
		end;

		if ($@MSCount == 0) end;
		emotion 27;
		specialeffect 267;
		specialeffect 267;
		specialeffect 267;
		specialeffect 267;
		specialeffect 267;
		mapannounce "minesweep","[GO!]",1;
		set $@MSCD,0;
		set $@MSCount,0;
		set $@MSCYCLE,0;
		set $@MSTotal,0;
		setnpctimer 0;
		goto L_TurnOn;
	OnTimer30000:
		if ($@MSReady == 1 && $@MSLose == 0) mapannounce "minesweep","Please redeem your points with the Room Controller, you will be returned to town automatically in one minute.",1;
		if ($@MSLose == 1) mapannounce "minesweep","The game is over, you will be returned to town automatically in one minute.",1;
		end;
	OnTimer60000:
		if ($MSAutoRst == 0 && getmapusers("minesweep") == 0) goto L_GMReset;
		if ($@MSReady == 0) mapannounce "minesweep","Four minutes remaining.",1;
		if ($@MSReady == 1 && $@MSLose == 0) mapannounce "minesweep","Please redeem your points with the Room Controller, you will be returned to town automatically in thirty seconds.",1;
		if ($@MSLose == 1) mapannounce "minesweep","The game is over, you will be returned to town automatically in thirty seconds.",1;
		end;
	OnTimer70000:
		if ($@MSReady == 1 && $@MSLose == 0) mapannounce "minesweep","Please redeem your points with the Room Controller, you will be returned to town automatically in twenty seconds.",1;
		if ($@MSLose == 1) mapannounce "minesweep","You will be returned to town automatically in twenty seconds.",1;
		end;
	OnTimer80000:
		if ($@MSReady == 1 && $@MSLose == 0) mapannounce "minesweep","Please redeem your points with the Room Controller, you will be returned to town automatically in ten seconds.",1;
		if ($@MSLose == 1) mapannounce "minesweep","You will be returned to town automatically in ten seconds.",1;
		end;
	OnTimer85000:
		if ($@MSReady == 1 && $@MSLose == 0) mapannounce "minesweep","Please redeem your points with the Room Controller, you will be returned to town automatically in five seconds.",1;
		if ($@MSLose == 1) mapannounce "minesweep","You will be returned to town automatically in five seconds.",1;
		end;
	OnTimer86000:
		if ($@MSReady == 1) mapannounce "minesweep","Four!",1;
		end;
	OnTimer87000:
		if ($@MSReady == 1) mapannounce "minesweep","Three!",1;
		end;
	OnTimer88000:
		if ($@MSReady == 1) mapannounce "minesweep","Two!",1;
		end;
	OnTimer89000:
		if ($@MSReady == 1) mapannounce "minesweep","One!!",1;
		end;
	OnTimer90000:
		if ($MSAutoRst == 0 && getmapusers("minesweep") == 0) goto L_GMReset;
		if ($@MSReady == 1) goto L_ResetPC;
		mapannounce "minesweep","Three minutes remaining.",1;
		end;
	OnTimer120000:
		if ($MSAutoRst == 0 && getmapusers("minesweep") == 0) goto L_GMReset;
		mapannounce "minesweep","Two minutes remaining.",1;
		end;
	OnTimer180000:
		if ($MSAutoRst == 0 && getmapusers("minesweep") == 0) goto L_GMReset;
		if ($@MSReady == 0) mapannounce "minesweep","One minute remaining.",1;
		end;
	OnTimer210000:
		mapannounce "minesweep","Thirty seconds remaining.",1;
		end;
	OnTimer220000:
		mapannounce "minesweep","Twenty seconds remaining.",1;
		end;
	OnTimer230000:
		mapannounce "minesweep","Ten seconds remaining.",1;
		end;
	OnTimer235000:
		mapannounce "minesweep","Five!",1;
		end;
	OnTimer236000:
		mapannounce "minesweep","Four!",1;
		end;
	OnTimer237000:
		mapannounce "minesweep","Three!!",1;
		end;
	OnTimer238000:
		mapannounce "minesweep","Two!!",1;
		end;
	OnTimer239000:
		mapannounce "minesweep","One!!!",1;
		end;
	OnTimer240000:
		if ($MSAutoRst == 0 && getmapusers("minesweep") == 0) goto L_GMReset;
		mapannounce "minesweep","You have FAILED!!  Five minutes have elapsed and the game is over, the room will be able to start another game in 3 minutes.  You will be returned to town automatically in ninety seconds.",1;
		set $@MSReady,1;
		set $@MSLose,1;
		set $@MSCYCLE,0;
		set $@MSSequence,0;
		setnpctimer 0;
		goto L_TurnOff;
	L_ResetPC:
		set $@MSOn,0;
		set $@MSLose,0;
		set $@MSTitle$,"- Room in Cooldown";
		mapannounce "minesweep","Thank you for playing.",1;
		areawarp "minesweep",1,1,400,400,"prontera",173,183;
		end;
	L_Reset:
		set $@MSReady,0;
		set $@MSTitle$,"";
		StopNPCTimer "Minesweeper1";
		setnpctimer 0;
		mapannounce "minesweep","This room is now ready to start a game.",1;
		end;
	L_GMReset:
		set $@MSOn,0;
		set $@MSLose,0;
		set $@MSReady,0;
		set $@MSTitle$,"";
		cleararray $@MSMines[0],0,82;
		cleararray $@MSField[0],0,82;
		cleararray $@MSState[0],0,82;
		StopNPCTimer "Minesweeper1";
		setnpctimer 0;
		areawarp "minesweep",1,1,400,400,"prontera",173,183;
		goto L_TurnOff;
	L_TurnOn:
		if ($@MSCYCLE == 81 && $@MSDebug == 1) debugmes "81 Tiles loaded into grid";
		if ($@MSCYCLE == 81) goto L_StopCycle;
		set $@MSCYCLE,$@MSCYCLE + 1;
		enablenpc "Tile1_"+$@MSCYCLE+"";
		setnpcdisplay "Tile1_"+$@MSCYCLE," ",1113;
		donpcevent "Tile1_"+$@MSCYCLE+"::OnFlip";
		goto L_TurnOn;
	L_TurnOff:
		if ($@MSCYCLE == 81 && $@MSDebug == 1) debugmes "81 Tiles removed from grid";
		if ($@MSCYCLE == 81) goto L_StopCycle;
		set $@MSCYCLE,$@MSCYCLE + 1;
		delwaitingroom "Tile1_"+$@MSCYCLE;
		donpcevent "Tile1_"+$@MSCYCLE+"::OnFlip";
		disablenpc "Tile1_"+$@MSCYCLE+"";
		goto L_TurnOff;
	L_StopCycle:
		cleararray $@MSMines[0],0,82;
		cleararray $@MSState[0],0,82;
		set $@MSCYCLE,0;
		end;
//---GM Control Panel----------------------------------------------------------------------------------
	M_Control:
		next;
		mes "[Room Controller]";
		mes "What would you like to do?";
		next;
		if (getgmlevel() < 99 && @MSOn == 1) menu "Change Auto Reset",M_AutoReset,"Change Settings",-,"Change Reward",M_Reward,"Reset Room",M_RmReset,"Return to Menu",M_Menu;
		if (getgmlevel() < 99 && @MSLose == 1 || $@MSReady == 1) menu "Change Auto Reset",M_AutoReset,"Change Settings",-,"Change Reward",M_Reward,"Reset Room",M_RmReset,"Return to Menu",M_Menu;
		if (getgmlevel() < 99) menu "Change Auto Reset",M_AutoReset,"Change Settings",-,"Change Reward",M_Reward,"Reset Room",M_RmReset,"Return to Menu",M_Menu;
		if (getgmlevel() == 99 && @MSOn == 1) menu "Change Auto Reset",M_AutoReset,"Change Settings",-,"Change Reward",M_Reward,"Reset Room",M_RmReset,"Return to Menu",M_Menu,"Console Debug",M_Debug;
		if (getgmlevel() == 99 && @MSLose == 1 || $@MSReady == 1) menu "Change Auto Reset",M_AutoReset,"Change Settings",-,"Change Reward",M_Reward,"Reset Room",M_RmReset,"Return to Menu",M_Menu,"Console Debug",M_Debug;
		if (getgmlevel() == 99) menu "Change Auto Reset",M_AutoReset,"Change Settings",-,"Change Reward",M_Reward,"Reset Room",M_RmReset,"Return to Menu",M_Menu,"Console Debug",M_Debug;
	M_Settings:
		mes "[Room Controller]";
		mes "Please select a setting to alter.";
		next;
		menu "Number Display",-,"Change Boom",M_Boom,"Set Min/Max Mines",M_MinMax,"Set Pain",M_Pain,"Change Sequence Cap",M_Sequence,"Reset Room",M_RmReset,"Return",M_Control;
		mes "[Room Controller]";
		mes "This changes the number display.";
		mes "0 = Display Waiting Room, NPC has no name";
		mes "1 = NPC name displays number, no waiting room";
		mes "2 = Display Waiting Room, NPC's name displays the number";
		next;
	L_Input:
		input $MSWindow;
		next;
		if ($MSWindow < 0 || $MSWindow > 2) message strcharinfo(0),"Number must be 0, 1, or 2.";
		if ($MSWindow < 0 || $MSWindow > 2) goto L_Input;
		mes "[Room Controller]";
		mes "Setting changed to "+$MSWindow+".";
		next;
		goto M_Control;
	M_Sequence:
		mes "[Room Controller]";
		mes "This controls how much the reaction will fission, or how many chain reactions can fire one after another.  The higher the reaction number is, the more points players will be able to gain from empty tiles.  This feature is pointless on minefields with a high number of mines, as there are no empty tiles.";
		next;
		mes "[Room Controller]";
		mes "The setting cannot be higher then 15, or lower then 2.";
		mes " ";
		mes " ";
		mes "Current setting : "+$MSFission+".";
		next;
	L_Sequence:
		input $MSFission;
		next;
		if ($MSFission < 2 || $MSFission > 15) message strcharinfo(0),"Number must be between 2 and 15!";
		if ($MSFission < 2 || $MSFission > 15) goto L_Sequence;
		mes "[Room Controller]";
		mes "Thank you, the setting has been changed to "+$MSFission+".";
		next;
		goto M_Control;
	M_Boom:
		mes "[Room Controller]";
		if ($MSBoom == 0) mes "The chain explosion on a game over is currently on.  Turn off?";
		if (($MSBoom == 0) && select("Yes","No") == 1) set $MSBoom,1;
		if ($MSBoom == 1) mes "The chain explosion on a game over is currently off.  Turn on?";
		if (($MSBoom == 1) && select("Yes","No") == 1) set $MSBoom,0;
		next;
		goto M_Control;
	M_MinMax:
		mes "[Room Controller]";
		mes "Here you may set the minimum, and maximum, limitations of the mines players can designate per game.  Suggested range is 8-25 mines.";
		next;
		mes "[Room Controller]";
		mes "The minimum cannot be lower then 5, or higher then 60.";
		mes " ";
		mes "Please input the Minimum limit.";
		next;
	L_Minimum:
		input $MSMinimum;
		next;
		if ($MSMinimum < 5 || $MSMinimum > 60) message strcharinfo(0),"Number must be between 5 and 60!";
		if ($MSMinimum < 5 || $MSMinimum > 60) goto L_Minimum;
		mes "[Room Controller]";
		mes "The maximum cannot be lower then 15, or higher then 75.";
		mes " ";
		mes "Please input the Maximum limit.";
		next;
	L_Maximum:
		input $MSMaximum;
		next;
		if ($MSMaximum < 15 || $MSMaximum > 75) message strcharinfo(0),"Number must be between 15 and 75!";
		if ($MSMaximum < 15 || $MSMaximum > 75) goto L_Maximum;
		mes "[Room Controller]";
		mes "The mine range is now "+$MSMinimum+" to "+$MSMaximum+" mines.";
		next;
		goto M_Control;
	M_Pain:
		mes "[Room Controller]";
		mes "This controls whether or not to damage the player who clicks the losing mine, and how much.";
		next;
		if ($MSHurts == 2) menu "Turn Off",M_PainOff,"Change to Percent",M_PainPercent,"Change damage",-;
		if ($MSHurts == 1) menu "Turn Off",M_PainOff,"Change to Digits",M_PainDigit,"Change damage",-;
		if ($MSHurts == 0) menu "Turn On",M_PainOn;
	M_PainPercent:
		mes "[Room Controller]";
		mes "This damage is percentile based.  Set the number of hp damage between -1% and -100%.  (Input the number, not the - sign)";
		mes " ";
		if ($MSHurts == 1) mes "Current setting is "+$MSOw+"%.";
		next;
	L_Pain1Input:
		input $MSOw;
		next;
		if ($MSOw < 1 || $MSOw > 100) message strcharinfo(0),"Number must be between 1 and 100!";
		if ($MSOw < 1 || $MSOw > 100) goto L_Pain1Input;
		set $MSOw,($MSOw*-1);
		set $MSHurts,1;
		goto M_Control;
	M_PainDigit:
		mes "[Room Controller]";
		mes "This damage is number based.  Set the number of hp damage between -1 and -999999.  (Input the number, not the - sign)";
		mes " ";
		if ($MSHurts == 2) mes "Current setting is "+$MSOw+".";

//---Tile Dialogue----------------------------------------------------------------------------------
function	script	F_MTile	{
	set $@MSNAME$,strcharinfo(0);
	if ($@MSState[getarg(0)] == 2) end;
	if ($@MSState[getarg(0)] == 0) goto L_Flag;
	if ($@MSState[getarg(0)] < 2) set $@MSState[getarg(0)],2;
	if ($@MSField[getarg(0)] == 0) goto L_Chain;
	if ($@MSField[getarg(0)] == 9) goto L_Lose;
	if ($@MSDebug == 1) debugmes "Tile "+getarg(0)+" / Point Value "+$@MSField[getarg(0)];
	if ($MSWindow == 0) setnpcdisplay "Tile1_"+getarg(0)," ",1002;
	if ($MSWindow == 1) setnpcdisplay "Tile1_"+getarg(0),""+$@MSField[getarg(0)]+"",1002;
	if ($MSWindow == 0) waitingroom "----- "+$@MSField[getarg(0)]+" -----",0;
	if ($MSWindow == 2) setnpcdisplay "Tile1_"+getarg(0),""+$@MSField[getarg(0)]+"",1002;
	if ($MSWindow == 2) waitingroom "----- "+$@MSField[getarg(0)]+" -----",0;
	if ($MSState[getarg(0)] == 0) set $@MSState[getarg(0)],2;
	set $@MSTotal,$@MSTotal + 1;
	if ($@MSDebug == 1) debugmes "Victory Total "+$@MSTotal+"/"+(162-$@MSAmount)+", tile : "+getarg(0);
	donpcevent "Tile1_"+getarg(0)+"::OnFlip";
	set @MSpt,@MSpt + $@MSField[getarg(0)];
	dispbottom "[Points : "+@MSpt+"]";
	if ($@MSTotal == (162 - $@MSAmount)) donpcevent "Minesweeper1::OnWin";
	end;
L_Flag:
	set $@MSTotal,$@MSTotal + 1;
	if ($@MSDebug == 1) debugmes "Victory Total "+$@MSTotal+"/"+(162-$@MSAmount)+", tile : "+getarg(0);
	set $@MSState[getarg(0)],1;
	setnpcdisplay "Tile1_"+getarg(0),"?",1242;
	donpcevent "Tile1_"+getarg(0)+"::OnFlip";
	if ($@MSTotal == (162 - $@MSAmount)) donpcevent "Minesweeper1::OnWin";
	end;
L_Lose:
	if ($@MSDebug == 1) debugmes strcharinfo(0)+" triggered mine "+getarg(0);
	mapannounce "minesweep",strcharinfo(0)+" has hit a mine!! Game over!  You will be returned to town automatically in ninety seconds.",1;
	set $@MSReady,1;
	set $@MSTotal,0;
	set $@MSLose,1;
	set @MSpt,0;
	specialeffect 183;
	specialeffect 107;
	specialeffect 17;
	specialeffect 8;
	specialeffect 8;
	if ($MSHurts == 1) percentheal $MSOw,0;
	if ($MSHurts == 2) heal $MSOw,0;
	donpcevent "Minesweeper1::OnBoom";
	donpcevent "Tile1_"+getarg(0)+"::OnFlip";
	if ($@MSDebug == 1) debugmes "Tile "+getarg(0)+" exploded!";
	set $@MSCYCLE,0;
L_BoomCycle:
	if ($@MSCYCLE == 81) goto L_BoomStop;
	set $@MSCYCLE,$@MSCYCLE + 1;
	donpcevent "Tile1_"+$@MSCYCLE+"::OnFlip";
	disablenpc "Tile1_"+$@MSCYCLE;
	goto L_BoomCycle;
L_BoomStop:
	set $@MSCYCLE,0;
	end;
L_Chain:
	set $@MSTotal,$@MSTotal - 1;
	if ($@MSDebug == 1) debugmes "Victory Total "+$@MSTotal+"/"+(162-$@MSAmount)+", tile : "+getarg(0);
	set $@MSSequence,0;
	donpcevent "Tile1_"+getarg(0)+"::OnReaction";
	end;
}

	if ($@MSLose == 1) specialeffect 150;
	if ($@MSLose == 0 && $@Ready == 1) specialeffect 70;
	if ($@MSLose == 0 && $@Ready == 0) specialeffect 423;
	end;
}
