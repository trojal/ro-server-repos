	if (getarg(1)==1) mes "[Claire]";
	if (getarg(1)==2) mes "[Joyce]";
	if (getarg(1)==3) mes "[Mindy]";
	mes "How many would you like?";

	if (@PVPITEM>500) set @PVPITEM,500;

	set @PVPITEM,0;

function	script	F_PVPLADDER	{
	set @CYCLE,0;
L_CheckforName:
	if (@CYCLE==(getarraysize($PVPLADDER_PT[0])+1)) goto L_CheckLad;
	if ($PVPLADDER_NAME$[@CYCLE]==strcharinfo(0)) goto L_MoveName;
	set @CYCLE,@CYCLE+1;
	goto L_CheckforName;
L_MoveName:
	if (KillPts < $PVPLADDER_PT[@CYCLE]) end;
	copyarray $PVPLADDER_PT[@CYCLE],$PVPLADDER_PT[(@CYCLE+1)],1;
	copyarray $PVPLADDER_NAME$[@CYCLE],$PVPLADDER_NAME$[(@CYCLE+1)],1;
L_CheckLad:
	set @CYCLE,0;
L_CheckLadder:
	if (@CYCLE==(getarraysize($PVPLADDER_PT[0])+1)) end;
	if (KillPts > $PVPLADDER_PT[@CYCLE]) goto L_ChangeLadder;
	set @CYCLE,@CYCLE+1;
	goto L_CheckLadder;
L_ChangeLadder:
	copyarray $PVPLADDER_PT[(1+@CYCLE)],$PVPLADDER_PT[@CYCLE],1;
	copyarray $PVPLADDER_NAME$[(1+@CYCLE)],$PVPLADDER_NAME$[@CYCLE],1;
	set $PVPLADDER_PT[@CYCLE],KillPts;
	set $PVPLADDER_NAME$[@CYCLE],strcharinfo(0);
	if ($PVPLADDER_NAME$[@CYCLE]==$PVPLADDER_NAME$[(1+@CYCLE)]) goto L_NameDouble;
	set @CYCLE,0;
	end;
L_NameDouble:
	set $PVPLADDER_PT[(1+@CYCLE)],0;
	set $PVPLADDER_NAME$[(1+@CYCLE)],"";
	set @CYCLE,0;
	end;
}
